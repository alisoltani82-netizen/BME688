/**
 * BME688 Burn-In / Conditioning Script
 * 
 * Run this for 48 hours when you first receive the sensor.
 * After initial burn-in, run for 30 minutes before taking measurements.
 * 
 * This script:
 * - Continuously heats the gas sensor
 * - Tracks gas resistance baseline over time
 * - Shows when the sensor is stabilizing
 * 
 * Wiring (SPI):
 *   BME688 VCC  -> 3.3V
 *   BME688 GND  -> GND
 *   BME688 SCK  -> GPIO4
 *   BME688 SDI  -> GPIO6
 *   BME688 SDO  -> GPIO5
 *   BME688 CS   -> GPIO7
 */

#include <Arduino.h>
#include <SPI.h>
#include <Adafruit_Sensor.h>
#include <Adafruit_BME680.h>

// SPI pins for ESP32-C3 SuperMini
#define BME_SCK   4
#define BME_MOSI  6
#define BME_MISO  5
#define BME_CS    7

#define SEALEVELPRESSURE_HPA (1013.25)

Adafruit_BME680 bme(BME_CS, BME_MOSI, BME_MISO, BME_SCK);

// Tracking variables
unsigned long startTime = 0;
unsigned long readingCount = 0;
float minGasResistance = 999999.0;
float maxGasResistance = 0.0;
float lastGasResistance = 0.0;
float gasResistanceSum = 0.0;

// For stability detection
float stabilityWindow[10];
int stabilityIndex = 0;
bool windowFilled = false;

void printElapsedTime(unsigned long ms) {
    unsigned long seconds = ms / 1000;
    unsigned long minutes = seconds / 60;
    unsigned long hours = minutes / 60;
    
    Serial.print(hours);
    Serial.print("h ");
    Serial.print(minutes % 60);
    Serial.print("m ");
    Serial.print(seconds % 60);
    Serial.print("s");
}

float calculateStability() {
    if (!windowFilled) return 0;
    
    float sum = 0;
    float avg = 0;
    
    for (int i = 0; i < 10; i++) {
        avg += stabilityWindow[i];
    }
    avg /= 10;
    
    // Calculate variance
    for (int i = 0; i < 10; i++) {
        float diff = stabilityWindow[i] - avg;
        sum += diff * diff;
    }
    
    float variance = sum / 10;
    float stdDev = sqrt(variance);
    
    // Return coefficient of variation (lower = more stable)
    return (stdDev / avg) * 100.0;
}

void setup() {
    Serial.begin(115200);
    delay(3000);
    
    Serial.println();
    Serial.println("╔════════════════════════════════════════╗");
    Serial.println("║     BME688 BURN-IN / CONDITIONING      ║");
    Serial.println("╠════════════════════════════════════════╣");
    Serial.println("║  First use: Run for 48 hours           ║");
    Serial.println("║  Regular use: Run 30 min before use    ║");
    Serial.println("╚════════════════════════════════════════╝");
    Serial.println();

    if (!bme.begin()) {
        Serial.println("ERROR: BME688 not found! Check wiring.");
        while (1) delay(1000);
    }

    Serial.println("✓ BME688 sensor found!");
    
    // Configure for burn-in (continuous heating)
    bme.setTemperatureOversampling(BME680_OS_8X);
    bme.setHumidityOversampling(BME680_OS_2X);
    bme.setPressureOversampling(BME680_OS_4X);
    bme.setIIRFilterSize(BME680_FILTER_SIZE_3);
    bme.setGasHeater(320, 150); // 320°C for 150ms - standard heating
    
    Serial.println("✓ Sensor configured for burn-in");
    Serial.println();
    Serial.println("Starting burn-in process...");
    Serial.println("Gas resistance will slowly rise and stabilize.");
    Serial.println("────────────────────────────────────────");
    Serial.println();
    
    startTime = millis();
    
    // Initialize stability window
    for (int i = 0; i < 10; i++) {
        stabilityWindow[i] = 0;
    }
}

void loop() {
    if (!bme.performReading()) {
        Serial.println("Failed to read sensor!");
        delay(1000);
        return;
    }
    
    readingCount++;
    unsigned long elapsed = millis() - startTime;
    float gasResKohm = bme.gas_resistance / 1000.0;
    
    // Update tracking
    if (gasResKohm < minGasResistance) minGasResistance = gasResKohm;
    if (gasResKohm > maxGasResistance) maxGasResistance = gasResKohm;
    gasResistanceSum += gasResKohm;
    
    // Update stability window
    stabilityWindow[stabilityIndex] = gasResKohm;
    stabilityIndex = (stabilityIndex + 1) % 10;
    if (stabilityIndex == 0) windowFilled = true;
    
    float stability = calculateStability();
    float avgGasRes = gasResistanceSum / readingCount;
    
    // Print status
    Serial.println("┌─────────────────────────────────────────┐");
    Serial.print("│ Elapsed: ");
    printElapsedTime(elapsed);
    
    // Calculate target progress
    float progress48h = (elapsed / (48.0 * 60 * 60 * 1000)) * 100;
    float progress30m = (elapsed / (30.0 * 60 * 1000)) * 100;
    
    if (progress48h < 100) {
        Serial.print(" (");
        Serial.print(progress48h, 1);
        Serial.println("% of 48h)     │");
    } else {
        Serial.println(" ✓ 48h COMPLETE!    │");
    }
    
    Serial.println("├─────────────────────────────────────────┤");
    
    // Current readings
    Serial.print("│ Temp: ");
    Serial.print(bme.temperature, 1);
    Serial.print("°C  Humidity: ");
    Serial.print(bme.humidity, 1);
    Serial.println("%        │");
    
    Serial.print("│ Pressure: ");
    Serial.print(bme.pressure / 100.0, 1);
    Serial.println(" hPa                    │");
    
    Serial.println("├─────────────────────────────────────────┤");
    Serial.println("│ GAS SENSOR STATUS                       │");
    Serial.println("├─────────────────────────────────────────┤");
    
    Serial.print("│ Current:  ");
    Serial.print(gasResKohm, 2);
    Serial.println(" kΩ                      │");
    
    Serial.print("│ Average:  ");
    Serial.print(avgGasRes, 2);
    Serial.println(" kΩ                      │");
    
    Serial.print("│ Min/Max:  ");
    Serial.print(minGasResistance, 1);
    Serial.print(" / ");
    Serial.print(maxGasResistance, 1);
    Serial.println(" kΩ              │");
    
    // Stability indicator
    Serial.print("│ Stability: ");
    if (!windowFilled) {
        Serial.println("Collecting data...            │");
    } else if (stability < 2.0) {
        Serial.print(stability, 2);
        Serial.println("% ✓ STABLE               │");
    } else if (stability < 5.0) {
        Serial.print(stability, 2);
        Serial.println("% ◐ Stabilizing...        │");
    } else {
        Serial.print(stability, 2);
        Serial.println("% ○ Conditioning...       │");
    }
    
    Serial.println("├─────────────────────────────────────────┤");
    
    // Status message
    if (elapsed < 30 * 60 * 1000) {
        // Less than 30 minutes
        Serial.print("│ ⏳ Warm-up: ");
        Serial.print(30 - (elapsed / 60000));
        Serial.println(" min remaining            │");
    } else if (elapsed < 48 * 60 * 60 * 1000) {
        // Between 30 min and 48 hours
        Serial.println("│ ✓ Ready for basic readings              │");
        Serial.print("│ ⏳ Full burn-in: ");
        printElapsedTime(48UL * 60 * 60 * 1000 - elapsed);
        Serial.println(" left    │");
    } else {
        Serial.println("│ ✓✓ FULL BURN-IN COMPLETE!               │");
        Serial.println("│ Sensor is fully conditioned.            │");
    }
    
    Serial.print("│ Reading #");
    Serial.print(readingCount);
    Serial.println("                              │");
    Serial.println("└─────────────────────────────────────────┘");
    Serial.println();
    
    delay(10000); // Read every 10 seconds during burn-in
}
